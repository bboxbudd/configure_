# тест команды cp


ls


# Копируем файл hello.txt в docs/
cp /hello.txt /docs

# Проверим результат
ls /docs
cat /docs/hello.txt

# Копируем файл в новый путь
cp /data/numbers.txt /data/numbers_copy.txt
ls /data
cat /data/numbers_copy.txt
uniq /data/numbers_copy.txt

# Ошибка: источник не существует
cp /nofile.txt /data

# Ошибка: источник — директория
cp /docs /data

# Ошибка: каталог назначения не существует
cp /hello.txt /no_such_dir/abc.txt

# Проверим перезапись файла
cp /hello.txt /data/numbers_copy.txt
cat /data/numbers_copy.txt

exit
